#define SENSOR_PIN 32 // Pin conectado al sensor KY-017
#define LED_PIN 13    // Pin conectado al LED

void setup() {
  pinMode(SENSOR_PIN, INPUT);
  pinMode(LED_PIN, OUTPUT);
  
  Serial.begin(9600);
  Serial.println("Iniciando...");
}

void loop() {
  // Leer el estado del sensor KY-017
  int sensorValue = digitalRead(SENSOR_PIN);

  if (sensorValue == HIGH) {
    // Si el sensor est치 activado (mercurio inclinado), encender el LED
    digitalWrite(LED_PIN, HIGH);
    Serial.println("ON");
  } else {
    // Si el sensor no est치 activado, apagar el LED
    digitalWrite(LED_PIN, LOW);
    Serial.println("OFF");
  }

  delay(100); // Peque침a pausa para evitar la detecci칩n falsa
}
